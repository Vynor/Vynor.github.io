## B - 3^A ##
3进制，总是贪心使用高位，使得使用的数目最小
```c++
void solve() {
    vector<int> b(20);
    b[0] = 1;
    for (int i = 1; i <= 10; i ++ ) b[i] = b[i - 1] * 3;
    int m;
    cin >> m;
    vector<int> ans;
    for (int j = 10; j >= 0; j -- )
        if (b[j] <= m){
            int t = m / b[j];
            m -= t * b[j];
            while (t -- ) ans.push_back(j);
        }
    cout << ans.size() << '\n';
    for (auto x : ans)
        cout << x << ' ';
}
```
